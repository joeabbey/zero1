269/// Convert source text into a token stream (including a terminal EOF token).
270pub fn lex(source: &str) -> Vec<Token> {
271    let mut tokens = Vec::new();
272    let mut lexer = RawToken::lexer(source);
273    while let Some(raw) = lexer.next() {
274        let raw = raw.unwrap_or(RawToken::Error);
275        let span = lexer.span();
276        let token = Token {
277            kind: TokenKind::from(raw),
278            lexeme: lexer.slice().to_string(),
279            span: Span::new(span.start as u32, span.end as u32),
280        };
281        tokens.push(token);
282    }
283    tokens.push(Token {
284        kind: TokenKind::Eof,
285        lexeme: String::new(),
286        span: Span::new(source.len() as u32, source.len() as u32),
287    });
288    tokens
289}
