1//! Warning detection for effect checking.
2//!
3//! This module detects non-critical issues like:
4//! - Functions declare effects but don't use them
5//! - Over-permissive capability grants (caps not used)
6//! - Async functions with no async operations
7
8use crate::Effect;
9use std::collections::HashSet;
10use z1_ast::{FnDecl, Module, Span};
11
12/// A warning detected during effect checking.
13#[derive(Debug, Clone, PartialEq, Eq)]
14pub enum EffectWarning {
15    UnusedEffect {
16        effect: String,
17        function: String,
18        fn_span: Span,
19    },
20    UnusedCapability {
21        capability: String,
22        module: String,
23        module_span: Span,
24    },
25    AsyncWithoutAsync {
26        function: String,
27        fn_span: Span,
28    },
29}
30
31impl std::fmt::Display for EffectWarning {
32    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
33        match self {
34            EffectWarning::UnusedEffect {
35                effect, function, ..
36            } => {
37                write!(
38                    f,
39                    "Function '{function}' declares effect '{effect}' but may not use it"
40                )
41            }
42            EffectWarning::UnusedCapability {
43                capability, module, ..
44            } => {
45                write!(
46                    f,
47                    "Module '{module}' grants capability '{capability}' but no functions use it"
48                )
49            }
50            EffectWarning::AsyncWithoutAsync { function, .. } => {
51                write!(
52                    f,
53                    "Function '{function}' declares 'async' effect but contains no async operations"
54                )
55            }
56        }
57    }
58}
59
60impl EffectWarning {
61    pub fn span(&self) -> Span {
62        match self {
63            EffectWarning::UnusedEffect { fn_span, .. }
64            | EffectWarning::AsyncWithoutAsync { fn_span, .. } => *fn_span,
65            EffectWarning::UnusedCapability { module_span, .. } => *module_span,
66        }
67    }
68}

70/// Collect warnings from a module.
71pub fn collect_warnings(module: &Module) -> Vec<EffectWarning> {
72    let mut warnings = Vec::new();
73
74    // Collect all effects used by functions
75    let mut used_effects = HashSet::new();
76    for item in &module.items {
77        if let z1_ast::Item::Fn(fn_decl) = item {
78            for effect in &fn_decl.effects {
79                if let Some(eff) = Effect::parse(effect) {
80                    used_effects.insert(eff);
81                }
82            }
83        }
84    }
85
86    // Check for unused capabilities
87    let module_name = module.path.0.join(".");
88    for cap in &module.caps {
89        if let Some(base_cap) = parse_capability_base(cap) {
90            if !used_effects.contains(&base_cap) && base_cap != Effect::Pure {
91                warnings.push(EffectWarning::UnusedCapability {
92                    capability: cap.clone(),
93                    module: module_name.clone(),
94                    module_span: module.span,
95                });
96            }
97        }
98    }
99
100    // Check for unused effects in functions
101    for item in &module.items {
102        if let z1_ast::Item::Fn(fn_decl) = item {
103            warnings.extend(check_function_warnings(fn_decl));
104        }
105    }
106
107    warnings
108}

125/// Parse capability string to base effect.
126fn parse_capability_base(cap: &str) -> Option<Effect> {
127    if let Some((base, _suffix)) = cap.split_once('.') {
128        Effect::parse(base)
129    } else {
130        Effect::parse(cap)
131    }
132}

134/// Suggest a fix for unused capability.
135#[allow(dead_code)]
136pub fn suggest_unused_capability_fix(capability: &str, module: &str) -> String {
137    format!(
138        "Remove '{capability}' from module capabilities if not needed: module {module} caps=[...]"
139    )
140}

142/// Suggest a fix for unused effect.
143#[allow(dead_code)]
144pub fn suggest_unused_effect_fix(effect: &str) -> String {
145    format!("Remove '{effect}' from effect list if not used")
146}

189    #[test]
190    fn test_collect_warnings_no_unused() {
191        use z1_ast::{Block, Item, ModulePath, TypeExpr};
192
193        let fn_decl = FnDecl {
194            name: "test".to_string(),
195            params: vec![],
196            ret: TypeExpr::Path(vec!["Unit".to_string()]),
197            effects: vec!["net".to_string()],
198            body: Block::default(),
199            span: Span::new(0, 10),
200        };
201
202        let module = Module {
203            path: ModulePath::from_parts(vec!["test".to_string()]),
204            version: None,
205            ctx_budget: None,
206            caps: vec!["net".to_string()],
207            items: vec![Item::Fn(fn_decl)],
208            span: Span::new(0, 100),
209        };
210
211        let warnings = collect_warnings(&module);
212        // Should not warn about 'net' since it's used by the function
