45pub fn format_module(
46    module: &Module,
47    mode: Mode,
48    options: &FmtOptions,
49) -> Result<String, FmtError> {
50    let symbols = SymbolTable::new(module, options.symmap_style);
51    let mut formatter = Formatter::new(module, mode, symbols);
52    formatter.write_module_header();
53    formatter.write_items()?;
54    formatter.finish();
55    Ok(formatter.buf)
56}
57
58struct Formatter<'a> {
59    module: &'a Module,
60    mode: Mode,
61    buf: String,
62    symbols: SymbolTable,
63    sections_emitted: usize,
64}
65
66impl<'a> Formatter<'a> {
67    fn new(module: &'a Module, mode: Mode, symbols: SymbolTable) -> Self {
68        Self {
69            module,
70            mode,
71            buf: String::with_capacity(256),
72            symbols,
73            sections_emitted: 0,
74        }
75    }
76
77    fn finish(&mut self) {
78        if !self.buf.ends_with('\n') {
79            self.buf.push('\n');
80        }
81    }
82
83    fn write_module_header(&mut self) {
84        match self.mode {
85            Mode::Compact => self.write_compact_header(),
86            Mode::Relaxed => self.write_relaxed_header(),
87        }
88    }
89
90    fn write_items(&mut self) -> Result<(), FmtError> {
91        for item in &self.module.items {
92            match item {
93                Item::Import(import) => {
94                    self.section_break();
95                    self.write_import(import);
96                }
97                Item::Symbol(sym) => {
98                    self.section_break();
99                    self.write_symbol_map(sym);
100                }
101                Item::Type(ty) => {
102                    self.section_break();
103                    self.write_type_decl(ty);
104                }
105                Item::Fn(func) => {
106                    self.section_break();
107                    self.write_fn_decl(func);
108                }
109            }
110        }
111        Ok(())
112    }
113
114    fn write_compact_header(&mut self) {
115        self.buf.push_str("m ");
116        self.buf.push_str(&self.module_path());
117        if let Some(version) = &self.module.version {
118            self.buf.push(':');
119            self.buf.push_str(version);
120        }
121        if let Some(ctx) = self.module.ctx_budget {
122            self.buf.push(' ');
123            self.buf.push_str("ctx=");
124            self.buf.push_str(&ctx.to_string());
125        }
126        if !self.module.caps.is_empty() {
127            self.buf.push(' ');
128            self.buf.push_str("caps=");
129            self.buf.push('[');
130            self.buf.push_str(&self.module.caps.join(","));
131            self.buf.push(']');
132        }
133        self.buf.push('\n');
134    }
135
136    fn write_relaxed_header(&mut self) {
137        self.buf.push_str("module ");
138        self.buf.push_str(&self.module_path());
139        if let Some(version) = &self.module.version {
140            self.buf.push_str(" : ");
141            self.buf.push_str(version);
142        }
143        self.buf.push('\n');
144        if let Some(ctx) = self.module.ctx_budget {
145            self.buf.push_str("  ctx = ");
146            self.buf.push_str(&ctx.to_string());
147            self.buf.push('\n');
148        }
149        if !self.module.caps.is_empty() {
150            self.buf.push_str("  caps = [");
151            self.buf.push_str(&self.module.caps.join(", "));
152            self.buf.push_str("]\n");
153        }
154    }
155
156    fn write_import(&mut self, import: &Import) {
157        match self.mode {
158            Mode::Compact => {
159                self.buf.push_str("u ");
160            }
161            Mode::Relaxed => {
162                self.buf.push_str("use ");
163            }
164        }
165        self.buf.push('"');
166        self.buf.push_str(&import.path);
167        self.buf.push('"');
168        if let Some(alias) = &import.alias {
169            let alias_display = self.symbols.display_ident(alias, self.mode);
170            self.buf.push_str(" as ");
171            self.buf.push_str(&alias_display);
172        }
173        if !import.only.is_empty() {
174            self.buf.push_str(" only [");
175            self.buf.push_str(&import.only.join(", "));
176            self.buf.push(']');
177        }
178        self.buf.push('\n');
179    }
180
181    fn write_symbol_map(&mut self, map: &SymbolMap) {
182        let pairs = self.symbols.ordered_pairs(&map.pairs);
183        if pairs.is_empty() {
184            return;
185        }
186        match self.mode {
187            Mode::Compact => {
188                self.buf.push_str("#sym { ");
189                for (idx, (long, short)) in pairs.iter().enumerate() {
190                    if idx > 0 {
191                        self.buf.push_str(", ");
192                    }
193                    self.buf.push_str(long);
194                    self.buf.push_str(": ");
195                    self.buf.push_str(short);
196                }
197                self.buf.push_str(" }\n");
198            }
199            Mode::Relaxed => {
200                self.buf.push_str("// SymbolMap: { ");
201                for (idx, (long, short)) in pairs.iter().enumerate() {
202                    if idx > 0 {
203                        self.buf.push_str(", ");
204                    }
205                    self.buf.push_str(long);
206                    self.buf.push_str(" â†” ");
207                    self.buf.push_str(short);
208                }
209                self.buf.push_str(" }\n");
210                self.buf.push_str("#sym { ");
211                for (idx, (long, short)) in pairs.iter().enumerate() {
212                    if idx > 0 {
213                        self.buf.push_str(", ");
214                    }
215                    self.buf.push_str(long);
216                    self.buf.push_str(": ");
217                    self.buf.push_str(short);
218                }
219                self.buf.push_str(" }\n");
220            }
221        }
222    }
223
224    fn write_type_decl(&mut self, decl: &TypeDecl) {
225        match self.mode {
226            Mode::Compact => self.buf.push_str("t "),
227            Mode::Relaxed => self.buf.push_str("type "),
228        }
229        let name = self.symbols.display_ident(&decl.name, self.mode);
230        self.buf.push_str(&name);
231        self.buf.push_str(" = ");
232        self.buf.push_str(&self.format_type_expr(&decl.expr));
233        self.buf.push('\n');
234    }
235
236    fn write_fn_decl(&mut self, decl: &FnDecl) {
237        let kw = match self.mode {
238            Mode::Compact => "f",
239            Mode::Relaxed => "fn",
240        };
241        let name = self.symbols.display_ident(&decl.name, self.mode);
242        self.buf.push_str(kw);
243        self.buf.push(' ');
244        self.buf.push_str(&name);
245        self.buf.push('(');
246        let params = decl
247            .params
248            .iter()
249            .map(|param| self.format_param(param))
250            .collect::<Vec<_>>()
251            .join(", ");
252        self.buf.push_str(&params);
253        self.buf.push(')');
254        self.buf.push_str(match self.mode {
255            Mode::Compact => "->",
256            Mode::Relaxed => " -> ",
257        });
258        self.buf.push_str(&self.format_type_expr(&decl.ret));
259        if !decl.effects.is_empty() {
260            match self.mode {
261                Mode::Compact => {
262                    self.buf.push_str(" eff [");
263                }
264                Mode::Relaxed => {
265                    self.buf.push_str("\n  eff [");
266                }
267            }
268            self.buf.push_str(&decl.effects.join(", "));
269            self.buf.push(']');
270        }
271        write_block(self, &decl.body.raw);
272    }
273
274    fn format_param(&self, param: &Param) -> String {
275        let name = self.symbols.display_ident(&param.name, self.mode);
276        let ty = self.format_type_expr(&param.ty);
277        format!("{name}: {ty}")
278    }
279
280    fn format_type_expr(&self, expr: &TypeExpr) -> String {
281        match expr {
282            TypeExpr::Path(parts) => {
283                let segments = parts
284                    .iter()
285                    .map(|p| self.symbols.display_ident(p, self.mode))
286                    .collect::<Vec<_>>();
287                segments.join(".")
288            }
289            TypeExpr::Record(fields) => {
290                let inner = fields
291                    .iter()
292                    .map(|field| self.format_record_field(field))
293                    .collect::<Vec<_>>()
294                    .join(", ");
295                format!("{{ {inner} }}")
296            }
297        }
298    }
299
300    fn format_record_field(&self, field: &RecordField) -> String {
301        let name = self.symbols.display_ident(&field.name, self.mode);
302        let ty = self.format_type_expr(&field.ty);
303        format!("{name}: {ty}")
304    }
305
306    fn module_path(&self) -> String {
307        self.module
308            .path
309            .as_str_vec()
310            .iter()
311            .map(|segment| segment.as_str())
312            .collect::<Vec<_>>()
313            .join(".")
314    }
315
316    fn section_break(&mut self) {
317        if !self.buf.ends_with('\n') {
318            self.buf.push('\n');
319        }
320        if (self.sections_emitted > 0 || matches!(self.mode, Mode::Relaxed))
321            && !self.buf.ends_with("\n\n")
322        {
323            self.buf.push('\n');
324        }
325        self.sections_emitted += 1;
326    }
327}

329pub struct SymbolTable {
330    long_to_short: HashMap<String, String>,
331    short_to_long: HashMap<String, String>,
332    style: SymMapStyle,
333}
334
335impl SymbolTable {
336    fn new(module: &Module, style: SymMapStyle) -> Self {
337        let mut long_to_short = HashMap::new();
338        let mut short_to_long = HashMap::new();
339        for item in &module.items {
340            if let Item::Symbol(sym) = item {
341                for pair in &sym.pairs {
342                    long_to_short.insert(pair.long.clone(), pair.short.clone());
343                    short_to_long.insert(pair.short.clone(), pair.long.clone());
344                }
345            }
346        }
347        Self {
348            long_to_short,
349            short_to_long,
350            style,
351        }
352    }
353
354    /// Create a SymbolTable from a raw SymbolMap (for use in parser)
355    pub fn from_symbol_map(sym_map: &SymbolMap) -> Self {
356        let mut long_to_short = HashMap::new();
357        let mut short_to_long = HashMap::new();
358        for pair in &sym_map.pairs {
359            long_to_short.insert(pair.long.clone(), pair.short.clone());
360            short_to_long.insert(pair.short.clone(), pair.long.clone());
361        }
362        Self {
363            long_to_short,
364            short_to_long,
365            style: SymMapStyle::Respect,
366        }
367    }
368
369    /// Normalize an identifier to its canonical long form (for parser use)
370    pub fn normalize_ident(&self, ident: &str) -> String {
371        self.short_to_long
372            .get(ident)
373            .cloned()
374            .unwrap_or_else(|| ident.to_string())
375    }
376
377    fn display_ident(&self, ident: &str, mode: Mode) -> String {
378        match mode {
379            Mode::Compact => self
380                .long_to_short
381                .get(ident)
382                .cloned()
383                .unwrap_or_else(|| ident.to_string()),
384            Mode::Relaxed => self
385                .short_to_long
386                .get(ident)
387                .cloned()
388                .unwrap_or_else(|| ident.to_string()),
389        }
390    }
391
392    fn ordered_pairs<'a>(&'a self, pairs: &'a [SymbolPair]) -> Vec<(String, String)> {
393        let mut local = pairs
394            .iter()
395            .map(|p| (p.long.clone(), p.short.clone()))
396            .collect::<Vec<_>>();
397        match self.style {
398            SymMapStyle::Respect => local,
399            SymMapStyle::Reflow => {
400                local.sort_by(|a, b| a.0.cmp(&b.0));
401                local
402            }
403        }
404    }
405}
406fn write_block(formatter: &mut Formatter<'_>, raw: &str) {
407    // NOTE: Function bodies are NOT transformed by SymbolMap - they are semantic content.
408    // Only top-level declarations (module, type, function names) are transformed.
409    // This ensures identifiers in function bodies contribute to semantic hash.
410
411    if matches!(formatter.mode, Mode::Compact) {
412        formatter.buf.push(' ');
413        formatter.buf.push_str(raw);
414        formatter.buf.push('\n');
415        return;
416    }
417
418    // For relaxed mode, reformat the block with proper indentation
419    formatter.buf.push('\n');
420    let trimmed = raw.trim();
421
422    // Remove outer braces if present
423    let content = if trimmed.starts_with('{') && trimmed.ends_with('}') {
424        &trimmed[1..trimmed.len() - 1]
425    } else {
426        trimmed
427    };
428
429    formatter.buf.push_str("{\n");
430
431    // Process each line, tracking brace depth for proper indentation
432    let mut indent_level: usize = 1; // Start at 1 because we're inside the function body
433    for line in content.lines() {
434        let trimmed_line = line.trim();
435        if trimmed_line.is_empty() {
436            continue;
437        }
438
439        // Decrease indent before printing if line starts with }
440        if trimmed_line.starts_with('}') {
441            indent_level = indent_level.saturating_sub(1);
442        }
443
444        // Add indentation
445        for _ in 0..indent_level {
446            formatter.buf.push_str("  ");
447        }
448        formatter.buf.push_str(trimmed_line);
449        formatter.buf.push('\n');
450
451        // Increase indent after printing if line ends with {
452        if trimmed_line.ends_with('{') {
453            indent_level += 1;
454        }
455    }
456
457    formatter.buf.push_str("}\n");
