1//! Context estimator for Zero1 cells and functions.
2//!
3//! This crate provides token usage estimation for Z1 modules (cells) and functions,
4//! enabling enforcement of context budget limits defined in module headers.
5//!
6//! ## Token Cost Model
7//!
8//! The MVP uses a naive heuristic: `tokens ≈ ceil(chars / 3.8)`.
9//! Future versions will support model-specific dictionaries (SDict) for improved estimation.
10//!
11//! ## Usage
12//!
13//! ```rust
14//! use z1_ctx::estimate_cell;
15//! use z1_parse::parse_module;
16//!
17//! let source = r#"
18//! m http.server:1.0 ctx=128 caps=[net]
19//! f handler()->Unit eff [pure] { ret Unit }
20//! "#;
21//!
22//! let module = parse_module(source).unwrap();
23//! let estimate = estimate_cell(&module).unwrap();
24//! println!("Estimated tokens: {}", estimate.total_tokens);
25//! ```
26
27use std::fmt;
28use thiserror::Error;
29use z1_ast::{FnDecl, Module, Span};
30use z1_fmt::{format_module, FmtOptions, Mode};

32/// Default token cost model: tokens ≈ ceil(chars / 3.8)
33pub const DEFAULT_CHARS_PER_TOKEN: f64 = 3.8;
34
35/// Errors that can occur during context estimation.
36#[derive(Debug, Error)]
37pub enum CtxError {
38    #[error("formatting error during estimation: {0}")]
39    Format(#[from] z1_fmt::FmtError),
40
41    #[error("cell exceeds context budget: {actual}/{budget} tokens. {suggestion}")]
42    BudgetExceeded {
43        actual: u32,
44        budget: u32,
45        suggestion: String,
46        span: Span,
47    },
48
49    #[error("function '{name}' exceeds context budget: {actual}/{budget} tokens")]
50    FnBudgetExceeded {
51        name: String,
52        actual: u32,
53        budget: u32,
54        span: Span,
55    },
56}
57
58/// Context estimation result for a cell.
59#[derive(Debug, Clone, PartialEq, Eq)]
60pub struct CellEstimate {
61    /// Total estimated tokens for the entire cell
62    pub total_tokens: u32,
63    /// Declared context budget from module header
64    pub budget: Option<u32>,
65    /// Per-function estimates
66    pub functions: Vec<FnEstimate>,
67    /// Character count of compact representation
68    pub char_count: usize,
69}
70
71/// Context estimation result for a single function.
72#[derive(Debug, Clone, PartialEq, Eq)]
73pub struct FnEstimate {
74    /// Function name
75    pub name: String,
76    /// Estimated tokens for this function
77    pub tokens: u32,
78    /// Character count for this function
79    pub chars: usize,
80    /// Source span
81    pub span: Span,
82}
83
84/// Configuration for context estimation.
85#[derive(Debug, Clone)]
86pub struct EstimateConfig {
87    /// Characters per token ratio (default: 3.8)
88    pub chars_per_token: f64,
89    /// Whether to enforce budget limits
90    pub enforce_budget: bool,
91}
92
93impl Default for EstimateConfig {
94    fn default() -> Self {
95        Self {
96            chars_per_token: DEFAULT_CHARS_PER_TOKEN,
97            enforce_budget: true,
98        }
99    }
100}

102/// Estimates token usage for a module using default configuration.
103///
104/// This function:
105/// 1. Formats the AST to compact text using the current SymbolMap
106/// 2. Counts characters and estimates tokens via naive heuristic
107/// 3. Validates against declared budget (if `ctx=N` is present)
108/// 4. Returns error if budget is exceeded, or Ok with estimate
109///
110/// # Errors
111///
112/// Returns `CtxError::BudgetExceeded` if the module declares a context budget
113/// and the estimated tokens exceed it.
114pub fn estimate_cell(module: &Module) -> Result<CellEstimate, CtxError> {
115    estimate_cell_with_config(module, &EstimateConfig::default())
116}
117
118/// Estimates token usage for a module with custom configuration.
119pub fn estimate_cell_with_config(
120    module: &Module,
121    config: &EstimateConfig,
122) -> Result<CellEstimate, CtxError> {
123    // Format to compact mode for token estimation
124    let compact_text = format_module(module, Mode::Compact, &FmtOptions::default())?;
125    let char_count = compact_text.len();
126
127    // Calculate total tokens using naive heuristic
128    let total_tokens = estimate_tokens_from_chars(char_count, config.chars_per_token);
129
130    // Estimate per-function tokens (approximate by line counting)
131    let functions = estimate_functions(module, config);
132
133    let estimate = CellEstimate {
134        total_tokens,
135        budget: module.ctx_budget,
136        functions,
137        char_count,
138    };
139
140    // Validate budget if enforcement is enabled
141    if config.enforce_budget {
142        if let Some(budget) = module.ctx_budget {
143            if total_tokens > budget {
144                return Err(CtxError::BudgetExceeded {
145                    actual: total_tokens,
146                    budget,
147                    suggestion: suggest_split(&estimate),
148                    span: module.span,
149                });
150            }
151        }
152    }
153
154    Ok(estimate)
155}

157/// Estimates tokens from character count using the configured ratio.
158fn estimate_tokens_from_chars(chars: usize, chars_per_token: f64) -> u32 {
159    (chars as f64 / chars_per_token).ceil() as u32
160}

162/// Estimates token usage for individual functions.
163///
164/// Note: This is approximate as we don't have full statement-level formatting yet.
165/// We estimate based on the size of the function body as a proportion of total content.
166fn estimate_functions(module: &Module, config: &EstimateConfig) -> Vec<FnEstimate> {
167    let mut estimates = Vec::new();
168
169    for item in &module.items {
170        if let z1_ast::Item::Fn(fn_decl) = item {
171            let fn_estimate = estimate_function(fn_decl, config);
172            estimates.push(fn_estimate);
173        }
174    }
175
176    estimates
177}

179/// Estimates tokens for a single function.
180fn estimate_function(fn_decl: &FnDecl, config: &EstimateConfig) -> FnEstimate {
181    // Rough approximation: use body raw text length as basis
182    // In future, we could format individual functions to compact mode
183    let body_len = fn_decl.body.raw.len();
184
185    // Add signature overhead (name + params + return type + effects)
186    let sig_overhead = fn_decl.name.len()
187        + fn_decl
188            .params
189            .iter()
190            .map(|p| p.name.len() + 5)
191            .sum::<usize>()
192        + 20; // rough overhead for syntax
193
194    let total_chars = body_len + sig_overhead;
195    let tokens = estimate_tokens_from_chars(total_chars, config.chars_per_token);
196
197    FnEstimate {
198        name: fn_decl.name.clone(),
199        tokens,
200        chars: total_chars,
201        span: fn_decl.span,
202    }
203}

205/// Suggests how to split a cell that exceeds its budget.
206fn suggest_split(estimate: &CellEstimate) -> String {
207    if estimate.functions.is_empty() {
208        return "Consider reducing the size of this cell.".to_string();
209    }
210
211    // Find largest function by token count
212    let largest_fn = estimate.functions.iter().max_by_key(|f| f.tokens).unwrap();
213
214    if estimate.functions.len() == 1 {
215        format!(
216            "Consider splitting function '{}' ({} tokens) into smaller functions.",
217            largest_fn.name, largest_fn.tokens
218        )
219    } else {
220        format!(
221            "Consider moving function '{}' ({} tokens) to a separate cell.",
222            largest_fn.name, largest_fn.tokens
223        )
224    }
225}

227impl fmt::Display for CellEstimate {
228    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
229        writeln!(f, "Cell Estimate:")?;
230        writeln!(f, "  Total tokens: {}", self.total_tokens)?;
231        if let Some(budget) = self.budget {
232            writeln!(f, "  Budget: {budget}")?;
233            let percentage = (self.total_tokens as f64 / budget as f64) * 100.0;
234            writeln!(f, "  Usage: {percentage:.1}%")?;
235        }
236        writeln!(f, "  Characters: {}", self.char_count)?;
237
238        if !self.functions.is_empty() {
239            writeln!(f, "\nFunction Estimates:")?;
240            for fn_est in &self.functions {
241                writeln!(
242                    f,
243                    "  - {}: {} tokens ({} chars)",
244                    fn_est.name, fn_est.tokens, fn_est.chars
245                )?;
246            }
247        }
248
249        Ok(())
250    }
251}

253#[cfg(test)]
254mod tests {
255    use super::*;
256
257    #[test]
258    fn test_estimate_tokens_from_chars() {
259        // Default ratio: 3.8 chars per token
260        assert_eq!(estimate_tokens_from_chars(0, DEFAULT_CHARS_PER_TOKEN), 0);
261        assert_eq!(estimate_tokens_from_chars(3, DEFAULT_CHARS_PER_TOKEN), 1);
262        assert_eq!(estimate_tokens_from_chars(4, DEFAULT_CHARS_PER_TOKEN), 2);
263        assert_eq!(estimate_tokens_from_chars(38, DEFAULT_CHARS_PER_TOKEN), 10);
264        assert_eq!(estimate_tokens_from_chars(100, DEFAULT_CHARS_PER_TOKEN), 27);
265    }
266
267    #[test]
268    fn test_estimate_tokens_custom_ratio() {
269        // Custom ratio: 4.0 chars per token
270        assert_eq!(estimate_tokens_from_chars(100, 4.0), 25);
271        assert_eq!(estimate_tokens_from_chars(17, 4.0), 5);
272    }
