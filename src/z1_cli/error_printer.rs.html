1//! Pretty error printing with source context and colors.
2//!
3//! This module provides utilities for formatting compiler errors with:
4//! - File path and line:column location
5//! - Source snippet showing the relevant line(s)
6//! - Color-coded output (red for errors, yellow for warnings)
7//! - Caret/underline pointing to exact location
8//! - Optional help hints
9
10use colored::*;
11use std::env;
12use z1_ast::Span;
13use z1_effects::EffectError;
14use z1_parse::ParseError;
15use z1_typeck::TypeError;
16
17/// Configuration for error printing.
18#[derive(Debug, Clone)]
19pub struct ErrorPrinterConfig {
20    /// Enable colored output (respects NO_COLOR environment variable).
21    pub use_colors: bool,
22}
23
24impl Default for ErrorPrinterConfig {
25    fn default() -> Self {
26        Self {
27            use_colors: env::var("NO_COLOR").is_err(),
28        }
29    }
30}

32/// Pretty-print a parse error with source context.
33pub fn print_parse_error(
34    error: &ParseError,
35    source: &str,
36    file_path: &str,
37    config: &ErrorPrinterConfig,
38) {
39    let span = match error {
40        ParseError::Unexpected { span, .. } | ParseError::Invalid { span, .. } => *span,
41    };
42
43    let header = format!("Error: {error}");
44    let colored_header = if config.use_colors {
45        header.red().bold().to_string()
46    } else {
47        header
48    };
49
50    eprintln!("{colored_header}");
51    eprint_source_snippet(source, file_path, span, config);
52    eprintln!();
53}

55/// Pretty-print a type error with source context.
56pub fn print_type_error(
57    error: &TypeError,
58    source: &str,
59    file_path: &str,
60    config: &ErrorPrinterConfig,
61) {
62    let span_opt = match error {
63        TypeError::Mismatch { span, .. }
64        | TypeError::UndefinedType { span, .. }
65        | TypeError::UndefinedFunction { span, .. }
66        | TypeError::UndefinedVariable { span, .. }
67        | TypeError::ArityMismatch { span, .. } => Some(*span),
68        _ => None,
69    };
70
71    let header = format!("Type Error: {error}");
72    let colored_header = if config.use_colors {
73        header.red().bold().to_string()
74    } else {
75        header
76    };
77
78    eprintln!("{colored_header}");
79    if let Some(span) = span_opt {
80        eprint_source_snippet(source, file_path, span, config);
81    }
82    eprintln!();
83}

85/// Pretty-print an effect error with source context.
86pub fn print_effect_error(
87    error: &EffectError,
88    source: &str,
89    file_path: &str,
90    config: &ErrorPrinterConfig,
91) {
92    let span = match error {
93        EffectError::MissingCapability { fn_span, .. } => *fn_span,
94        EffectError::UnknownEffect { fn_span, .. } => *fn_span,
95    };
96
97    let header = format!("Effect Error: {error}");
98    let colored_header = if config.use_colors {
99        header.red().bold().to_string()
100    } else {
101        header
102    };
103
104    eprintln!("{colored_header}");
105    eprint_source_snippet(source, file_path, span, config);
106
107    // Add helpful hint for missing capability errors
108    if let EffectError::MissingCapability { effect, module, .. } = error {
109        let hint =
110            format!("Help: Add '{effect}' to module capabilities: module {module} caps=[{effect}]");
111        let colored_hint = if config.use_colors {
112            hint.yellow().to_string()
113        } else {
114            hint
115        };
116        eprintln!("{colored_hint}");
117    }
118    eprintln!();
119}

121/// Print a source snippet with location marker to stderr.
122fn eprint_source_snippet(source: &str, file_path: &str, span: Span, config: &ErrorPrinterConfig) {
123    let (line_num, col_num, line_text) = extract_line_info(source, span);
124
125    // Print location header: "  ┌─ file.z1c:5:12"
126    let location = format!("  ┌─ {file_path}:{line_num}:{col_num}");
127    let colored_location = if config.use_colors {
128        location.blue().to_string()
129    } else {
130        location
131    };
132    eprintln!("{colored_location}");
133    eprintln!("  │");
134
135    // Print line number and source line: " 5 │     let x: U32 = \"hello\";"
136    let line_num_str = format!("{line_num:>3}");
137    let colored_line_num = if config.use_colors {
138        line_num_str.blue().to_string()
139    } else {
140        line_num_str
141    };
142    eprintln!("{colored_line_num} │ {line_text}");
143
144    // Print caret line: "    │            ^^^"
145    let caret_offset = col_num - 1; // Column is 1-indexed
146    let span_len = (span.end - span.start).max(1) as usize;
147    let carets = "^".repeat(span_len);
148    let colored_carets = if config.use_colors {
149        carets.red().bold().to_string()
150    } else {
151        carets
152    };
153    eprintln!("    │ {}{colored_carets}", " ".repeat(caret_offset));
154}

156/// Extract line number, column number, and line text for a given span.
157fn extract_line_info(source: &str, span: Span) -> (usize, usize, String) {
158    let start_offset = span.start as usize;
159
160    // Find line number and column
161    let mut line_num = 1;
162    let mut col_num = 1;
163    let mut line_start_offset = 0;
164
165    for (offset, ch) in source.char_indices() {
166        if offset == start_offset {
167            break;
168        }
169        if ch == '\n' {
170            line_num += 1;
171            col_num = 1;
172            line_start_offset = offset + 1;
173        } else {
174            col_num += 1;
175        }
176    }
177
178    // Extract the line text
179    let line_end_offset = source[line_start_offset..]
180        .find('\n')
181        .map(|pos| line_start_offset + pos)
182        .unwrap_or(source.len());
183
184    let line_text = source[line_start_offset..line_end_offset].to_string();
185
186    (line_num, col_num, line_text)
187}
