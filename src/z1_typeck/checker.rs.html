20    /// Type check a complete module.
21    pub fn check_module(&mut self, module: &Module) -> TypeResult<()> {
22        // Set capabilities from module header
23        self.context.set_capabilities(module.caps.clone());
24
25        // First pass: collect all type and function declarations
26        for item in &module.items {
27            match item {
28                Item::Type(type_decl) => {
29                    self.collect_type_decl(type_decl)?;
30                }
31                Item::Fn(fn_decl) => {
32                    self.collect_function_signature(fn_decl)?;
33                }
34                Item::Import(import) => {
35                    self.process_import(import)?;
36                }
37                Item::Symbol(_) => {
38                    // Symbol maps are formatting-only, ignored for type checking
39                }
40            }
41        }
42
43        // Second pass: type check function bodies and verify types
44        for item in &module.items {
45            if let Item::Fn(fn_decl) = item {
46                self.check_function(fn_decl)?;
47            }
48        }
49
50        Ok(())
51    }

53    /// Collect a type declaration into the type environment.
54    fn collect_type_decl(&mut self, decl: &TypeDecl) -> TypeResult<()> {
55        let ty = Type::from_ast(&decl.expr);
56        self.type_env.define_type(decl.name.clone(), ty);
57        Ok(())
58    }

60    /// Collect a function signature into the context.
61    fn collect_function_signature(&mut self, decl: &FnDecl) -> TypeResult<()> {
62        // Check that required capabilities are granted
63        for effect in &decl.effects {
64            if let Some(cap) = effect_to_capability(effect) {
65                if !self.context.has_capability(cap) {
66                    return Err(TypeError::CapabilityNotGranted {
67                        cap: cap.to_string(),
68                    });
69                }
70            }
71        }
72
73        let param_types: Vec<Type> = decl
74            .params
75            .iter()
76            .map(|p| self.resolve_type(&p.ty, decl.span))
77            .collect::<TypeResult<Vec<_>>>()?;
78
79        let ret_type = self.resolve_type(&decl.ret, decl.span)?;
80
81        let func_type = Type::Function {
82            params: param_types,
83            ret: Box::new(ret_type),
84            effects: decl.effects.clone(),
85        };
86
87        self.context.define_function(decl.name.clone(), func_type);
88        Ok(())
89    }

91    /// Process an import statement (stubbed for MVP - full resolution requires module system).
92    fn process_import(&mut self, import: &Import) -> TypeResult<()> {
93        // For MVP, we register the alias if present
94        if let Some(alias) = &import.alias {
95            // We don't have the full module system yet, so we just track the alias
96            // In a complete implementation, this would load the imported module and
97            // resolve the types from it
98            let import_path = import.path.split('/').map(|s| s.to_string()).collect();
99            self.type_env.register_alias(alias.clone(), import_path);
100        }
101
102        // For now, we stub imported types as Path types
103        // A full implementation would resolve these from the imported module
104        for name in &import.only {
105            let qualified_name = if let Some(alias) = &import.alias {
106                vec![alias.clone(), name.clone()]
107            } else {
108                vec![name.clone()]
109            };
110
111            // Register as an opaque path type for now
112            let imported_type = Type::Path(qualified_name.clone());
113            self.type_env.register_import(qualified_name, imported_type);
114        }
115
116        Ok(())
117    }

119    /// Resolve a TypeExpr to a Type, handling path resolution.
120    fn resolve_type(&self, expr: &z1_ast::TypeExpr, _span: z1_ast::Span) -> TypeResult<Type> {
121        let ty = Type::from_ast(expr);
122
123        // If it's a path type, try to resolve it
124        if let Type::Path(ref path) = ty {
125            // Check if it's already a primitive
126            if ty.is_primitive() {
127                return Ok(ty);
128            }
129
130            // Try to look up in type environment
131            if let Some(resolved) = self.type_env.lookup(path) {
132                return Ok(resolved);
133            }
134
135            // If not found, it might be an imported type that we're treating as opaque
136            // For MVP, we allow path types to remain unresolved
137            Ok(ty)
138        } else {
139            Ok(ty)
140        }
141    }

143    /// Type check a function declaration.
144    fn check_function(&mut self, decl: &FnDecl) -> TypeResult<()> {
145        // Create a new context for this function scope
146        let mut func_ctx = self.context.enter_function(&decl.effects);
147
148        // Add parameters to the function context
149        for param in &decl.params {
150            let param_type = self.resolve_type(&param.ty, param.span)?;
151            func_ctx.define_variable(param.name.clone(), param_type);
152        }
153
154        // For MVP, we don't have full statement AST yet (body.raw is String)
155        // So we can't type check the function body in detail
156        // This is a known limitation documented in PROGRESS.md
157
158        // We do basic validation: check that the function signature is well-formed
159        let ret_type = self.resolve_type(&decl.ret, decl.span)?;
160
161        // Verify that the return type is valid
162        if let Type::Path(ref path) = ret_type {
163            if !ret_type.is_primitive() && self.type_env.lookup(path).is_none() {
164                // Allow unresolved paths for now (might be imported types)
165                // In a full implementation, this would be an error
166            }
167        }
168
169        Ok(())
170    }

172    /// Check that a call site's effects are compatible with the calling context.
173    pub fn check_effect_compatibility(
174        &self,
175        required_effects: &[String],
176        available_effects: &HashSet<String>,
177    ) -> TypeResult<()> {
178        for effect in required_effects {
179            if effect != "pure" && !available_effects.contains(effect) {
180                return Err(TypeError::EffectNotPermitted {
181                    effect: effect.clone(),
182                });
183            }
184        }
185        Ok(())
186    }

188    /// Check structural type equality (public for testing).
189    pub fn check_type_equality(
190        &self,
191        expected: &Type,
192        found: &Type,
193        span: z1_ast::Span,
194    ) -> TypeResult<()> {
195        if !expected.structural_eq(found) {
196            return Err(TypeError::mismatch(
197                expected.display_name(),
198                found.display_name(),
199                span,
200            ));
201        }
202        Ok(())
203    }

205    /// Check function call arity and types (public for testing).
206    pub fn check_call(
207        &self,
208        func_type: &Type,
209        args: &[Type],
210        span: z1_ast::Span,
211    ) -> TypeResult<Type> {
212        match func_type {
213            Type::Function {
214                params,
215                ret,
216                effects,
217            } => {
218                // Check arity
219                if params.len() != args.len() {
220                    return Err(TypeError::arity_mismatch(params.len(), args.len(), span));
221                }
222
223                // Check parameter types
224                for (param_ty, arg_ty) in params.iter().zip(args.iter()) {
225                    if !param_ty.structural_eq(arg_ty) {
226                        return Err(TypeError::mismatch(
227                            param_ty.display_name(),
228                            arg_ty.display_name(),
229                            span,
230                        ));
231                    }
232                }
233
234                // Check effects
235                self.check_effect_compatibility(effects, self.context.available_effects())?;
236
237                Ok((**ret).clone())
238            }
239            _ => Err(TypeError::Mismatch {
240                expected: "function type".to_string(),
241                found: func_type.display_name(),
242                span,
243            }),
244        }
245    }
