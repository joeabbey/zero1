1use crate::types::Type;
2use std::collections::{HashMap, HashSet};
3use z1_ast::Ident;
4
5/// Typing context for variables, functions, and effects during type checking.
6pub struct Context {
7    /// Function signatures (name -> type)
8    functions: HashMap<Ident, Type>,
9
10    /// Variable types in current scope (name -> type)
11    variables: HashMap<Ident, Type>,
12
13    /// Effects available in the current context
14    available_effects: HashSet<Ident>,
15
16    /// Capabilities granted by the module
17    granted_capabilities: HashSet<String>,
18}
19
20impl Context {
21    pub fn new() -> Self {
22        Self {
23            functions: HashMap::new(),
24            variables: HashMap::new(),
25            available_effects: HashSet::new(),
26            granted_capabilities: HashSet::new(),
27        }
28    }
29
30    /// Register a function with its type signature.
31    pub fn define_function(&mut self, name: Ident, ty: Type) {
32        self.functions.insert(name, ty);
33    }
34
35    /// Lookup a function's type signature.
36    pub fn lookup_function(&self, name: &Ident) -> Option<&Type> {
37        self.functions.get(name)
38    }
39
40    /// Register a variable in the current scope.
41    pub fn define_variable(&mut self, name: Ident, ty: Type) {
42        self.variables.insert(name, ty);
43    }
44
45    /// Lookup a variable's type.
46    pub fn lookup_variable(&self, name: &Ident) -> Option<&Type> {
47        self.variables.get(name)
48    }
49
50    /// Add an effect to the available effects set.
51    pub fn add_effect(&mut self, effect: Ident) {
52        self.available_effects.insert(effect);
53    }
54
55    /// Check if an effect is available in the current context.
56    pub fn has_effect(&self, effect: &Ident) -> bool {
57        self.available_effects.contains(effect)
58    }
59
60    /// Get all available effects.
61    pub fn available_effects(&self) -> &HashSet<Ident> {
62        &self.available_effects
63    }
64
65    /// Set the capabilities granted by the module.
66    pub fn set_capabilities(&mut self, caps: Vec<String>) {
67        self.granted_capabilities = caps.into_iter().collect();
68    }
69
70    /// Check if a capability is granted.
71    pub fn has_capability(&self, cap: &str) -> bool {
72        self.granted_capabilities.contains(cap)
73    }
74
75    /// Create a new context inheriting functions and capabilities but with empty variables.
76    /// This is useful when entering a new function scope.
77    pub fn enter_function(&self, effects: &[Ident]) -> Self {
78        let mut ctx = Self {
79            functions: self.functions.clone(),
80            variables: HashMap::new(),
81            available_effects: effects.iter().cloned().collect(),
82            granted_capabilities: self.granted_capabilities.clone(),
83        };
84
85        // Pure functions can always be called
86        ctx.available_effects.insert("pure".to_string());
87
88        ctx
89    }
90
91    /// Enter a nested block scope (inherits everything from parent).
92    pub fn enter_block(&self) -> Self {
93        Self {
94            functions: self.functions.clone(),
95            variables: self.variables.clone(),
96            available_effects: self.available_effects.clone(),
97            granted_capabilities: self.granted_capabilities.clone(),
98        }
99    }
100}
101
102impl Default for Context {
103    fn default() -> Self {
104        Self::new()
105    }
106}
107
108/// Map effects to required capabilities.
109pub fn effect_to_capability(effect: &str) -> Option<&'static str> {
110    match effect {
111        "net" => Some("net"),
112        "fs" => Some("fs"),
113        "time" => Some("time"),
114        "crypto" => Some("crypto"),
115        "env" => Some("env"),
116        "unsafe" => Some("unsafe"),
117        "pure" => None,  // Pure effects require no capabilities
118        "async" => None, // Async is a language feature, not a capability
119        _ => None,
120    }
121}

123#[cfg(test)]

127    #[test]
128    fn test_function_definition_and_lookup() {
129        let mut ctx = Context::new();
130        let func_type = Type::Function {
131            params: vec![Type::U32],
132            ret: Box::new(Type::Bool),
133            effects: vec!["pure".to_string()],
134        };
135
136        ctx.define_function("test_fn".to_string(), func_type.clone());
137        assert_eq!(
138            ctx.lookup_function(&"test_fn".to_string()),
139            Some(&func_type)
140        );
141        assert_eq!(ctx.lookup_function(&"missing".to_string()), None);
142    }
143
144    #[test]
145    fn test_effect_tracking() {
146        let mut ctx = Context::new();
147        assert!(!ctx.has_effect(&"net".to_string()));
148
149        ctx.add_effect("net".to_string());
150        assert!(ctx.has_effect(&"net".to_string()));
151    }
152
153    #[test]
154    fn test_capability_tracking() {
155        let mut ctx = Context::new();
156        ctx.set_capabilities(vec!["net".to_string(), "time".to_string()]);
157
158        assert!(ctx.has_capability("net"));
159        assert!(ctx.has_capability("time"));
160        assert!(!ctx.has_capability("fs"));
161    }
162
163    #[test]
164    fn test_enter_function_scope() {
165        let mut ctx = Context::new();
166        ctx.define_function("outer".to_string(), Type::Bool);
167        ctx.define_variable("x".to_string(), Type::U32);
168        ctx.add_effect("net".to_string());
169
170        let func_ctx = ctx.enter_function(&["fs".to_string()]);
171
172        // Functions are inherited
173        assert!(func_ctx.lookup_function(&"outer".to_string()).is_some());
174
175        // Variables are NOT inherited (new scope)
176        assert!(func_ctx.lookup_variable(&"x".to_string()).is_none());
177
178        // Effects are replaced with new ones
179        assert!(!func_ctx.has_effect(&"net".to_string()));
180        assert!(func_ctx.has_effect(&"fs".to_string()));
181        assert!(func_ctx.has_effect(&"pure".to_string())); // pure always available
182    }
183
184    #[test]
185    fn test_effect_capability_mapping() {
186        assert_eq!(effect_to_capability("net"), Some("net"));
187        assert_eq!(effect_to_capability("fs"), Some("fs"));
188        assert_eq!(effect_to_capability("pure"), None);
189        assert_eq!(effect_to_capability("async"), None);
190    }
