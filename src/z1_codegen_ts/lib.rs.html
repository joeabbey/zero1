1//! TypeScript Code Generator for Zero1
2//!
3//! This crate generates TypeScript code from Zero1 IR. It provides a clean,
4//! idiomatic TypeScript output that can be used in Node.js or browser environments.
5
6use z1_ir::*;
7
8/// TypeScript code generator
9pub struct TsCodegen {
10    output: String,
11    indent_level: usize,
12}
13
14impl TsCodegen {
15    /// Create a new TypeScript code generator
16    pub fn new() -> Self {
17        TsCodegen {
18            output: String::new(),
19            indent_level: 0,
20        }
21    }
22
23    /// Generate TypeScript code from IR module
24    pub fn generate(&mut self, module: &IrModule) -> String {
25        self.output.clear();
26        self.indent_level = 0;
27
28        // File header comment
29        self.write_line("// Generated by Zero1 compiler");
30        self.write_line(&format!(
31            "// TypeScript output from module: {}",
32            module.name
33        ));
34        self.write_line(&format!("// Version: {}", module.version));
35        self.write_line("");
36
37        // Imports
38        for import in &module.imports {
39            self.gen_import(import);
40        }
41        if !module.imports.is_empty() {
42            self.write_line("");
43        }
44
45        // Type definitions
46        for type_def in &module.types {
47            self.gen_type_def(type_def);
48            self.write_line("");
49        }
50
51        // Functions
52        for func in &module.functions {
53            self.gen_function(func);
54            self.write_line("");
55        }
56
57        // Exports
58        if !module.exports.is_empty() {
59            self.write_line(&format!("export {{ {} }};", module.exports.join(", ")));
60        }
61
62        self.output.clone()
63    }
64
65    fn gen_import(&mut self, import: &IrImport) {
66        let items = import.items.join(", ");
67        let module_path = import.path.replace('/', "_");
68        if !items.is_empty() {
69            self.write_line(&format!("import {{ {items} }} from './{module_path}.js';"));
70        } else {
71            self.write_line(&format!("import './{module_path}.js';"));
72        }
73    }
74
75    fn gen_type_def(&mut self, type_def: &IrTypeDef) {
76        match &type_def.ty {
77            IrType::Record(fields) => {
78                self.write_line(&format!("export interface {} {{", type_def.name));
79                self.indent_level += 1;
80                for (field_name, field_type) in fields {
81                    let field_ty = self.type_to_ts(field_type);
82                    self.write_line(&format!("{field_name}: {field_ty};"));
83                }
84                self.indent_level -= 1;
85                self.write_line("}");
86            }
87            IrType::Union(variants) => {
88                let variant_types: Vec<String> = variants
89                    .iter()
90                    .map(|(name, ty)| {
91                        if let Some(inner) = ty {
92                            let inner_ts = self.type_to_ts(inner);
93                            format!("{{ tag: '{name}', value: {inner_ts} }}")
94                        } else {
95                            format!("{{ tag: '{name}' }}")
96                        }
97                    })
98                    .collect();
99                self.write_line(&format!(
100                    "export type {} = {};",
101                    type_def.name,
102                    variant_types.join(" | ")
103                ));
104            }
105            _ => {
106                let ty_ts = self.type_to_ts(&type_def.ty);
107                self.write_line(&format!("export type {} = {ty_ts};", type_def.name));
108            }
109        }
110    }
111
112    #[allow(clippy::only_used_in_recursion)]
113    fn type_to_ts(&self, ty: &IrType) -> String {
114        match ty {
115            IrType::Bool => "boolean".to_string(),
116            IrType::Str => "string".to_string(),
117            IrType::U16 | IrType::U32 | IrType::U64 => "number".to_string(),
118            IrType::Unit => "void".to_string(),
119            IrType::Named(name) => name.clone(),
120            IrType::Record(fields) => {
121                let field_strs: Vec<String> = fields
122                    .iter()
123                    .map(|(name, ty)| {
124                        let ty_ts = self.type_to_ts(ty);
125                        format!("{name}: {ty_ts}")
126                    })
127                    .collect();
128                format!("{{ {} }}", field_strs.join(", "))
129            }
130            IrType::Union(variants) => {
131                let variant_strs: Vec<String> = variants
132                    .iter()
133                    .map(|(name, ty)| {
134                        if let Some(inner) = ty {
135                            let inner_ts = self.type_to_ts(inner);
136                            format!("{{ tag: '{name}', value: {inner_ts} }}")
137                        } else {
138                            format!("{{ tag: '{name}' }}")
139                        }
140                    })
141                    .collect();
142                variant_strs.join(" | ")
143            }
144            IrType::Generic { base, args } => {
145                let arg_strs: Vec<String> = args.iter().map(|a| self.type_to_ts(a)).collect();
146                let base_ts = self.type_to_ts(base);
147                format!("{base_ts}<{}>", arg_strs.join(", "))
148            }
149        }
150    }
151
152    fn gen_function(&mut self, func: &IrFunction) {
153        // Function signature
154        let params: Vec<String> = func
155            .params
156            .iter()
157            .map(|(name, ty)| {
158                let ty_ts = self.type_to_ts(ty);
159                format!("{name}: {ty_ts}")
160            })
161            .collect();
162        let return_type = self.type_to_ts(&func.return_type);
163
164        // Check for async effect
165        let is_async = func
166            .effects
167            .iter()
168            .any(|e| e.contains("async") || e.contains("Async"));
169        let async_kw = if is_async { "async " } else { "" };
170
171        self.write_line(&format!(
172            "export {async_kw}function {}({}): {return_type} {{",
173            func.name,
174            params.join(", ")
175        ));
176
177        self.indent_level += 1;
178        self.gen_block(&func.body);
179        self.indent_level -= 1;
180
181        self.write_line("}");
182    }
183
184    fn gen_block(&mut self, block: &IrBlock) {
185        for stmt in &block.statements {
186            self.gen_stmt(stmt);
187        }
188    }
189
190    fn gen_stmt(&mut self, stmt: &IrStmt) {
191        match stmt {
192            IrStmt::Let {
193                name,
194                mutable,
195                ty,
196                value,
197            } => {
198                let var_kw = if *mutable { "let" } else { "const" };
199                let type_annotation = ty
200                    .as_ref()
201                    .map(|t| {
202                        let ty_ts = self.type_to_ts(t);
203                        format!(": {ty_ts}")
204                    })
205                    .unwrap_or_default();
206                let val_expr = self.gen_expr(value);
207                self.write_line(&format!("{var_kw} {name}{type_annotation} = {val_expr};"));
208            }
209            IrStmt::Assign { target, value } => {
210                let tgt = self.gen_expr(target);
211                let val = self.gen_expr(value);
212                self.write_line(&format!("{tgt} = {val};"));
213            }
214            IrStmt::If {
215                cond,
216                then_block,
217                else_block,
218            } => {
219                let cond_expr = self.gen_expr(cond);
220                self.write_line(&format!("if ({cond_expr}) {{"));
221                self.indent_level += 1;
222                self.gen_block(then_block);
223                self.indent_level -= 1;
224                if let Some(else_blk) = else_block {
225                    self.write_line("} else {");
226                    self.indent_level += 1;
227                    self.gen_block(else_blk);
228                    self.indent_level -= 1;
229                }
230                self.write_line("}");
231            }
232            IrStmt::While { cond, body } => {
233                let cond_expr = self.gen_expr(cond);
234                self.write_line(&format!("while ({cond_expr}) {{"));
235                self.indent_level += 1;
236                self.gen_block(body);
237                self.indent_level -= 1;
238                self.write_line("}");
239            }
240            IrStmt::Return { value } => {
241                if let Some(val) = value {
242                    let val_expr = self.gen_expr(val);
243                    self.write_line(&format!("return {val_expr};"));
244                } else {
245                    self.write_line("return;");
246                }
247            }
248            IrStmt::Expr(expr) => {
249                let expr_str = self.gen_expr(expr);
250                self.write_line(&format!("{expr_str};"));
251            }
252        }
253    }
254
255    fn gen_expr(&self, expr: &IrExpr) -> String {
256        match expr {
257            IrExpr::Var(name) => name.clone(),
258            IrExpr::Literal(lit) => self.gen_literal(lit),
259            IrExpr::BinOp { op, left, right } => {
260                let l = self.gen_expr(left);
261                let r = self.gen_expr(right);
262                let op_str = self.binop_to_ts(op);
263                format!("{l} {op_str} {r}")
264            }
265            IrExpr::UnaryOp { op, expr } => {
266                let op_str = self.unaryop_to_ts(op);
267                let expr_str = self.gen_expr(expr);
268                if *op == IrUnaryOp::Await {
269                    format!("{op_str} {expr_str}")
270                } else {
271                    format!("{op_str}{expr_str}")
272                }
273            }
274            IrExpr::Call { func, args } => {
275                let arg_strs: Vec<String> = args.iter().map(|a| self.gen_expr(a)).collect();
276                let func_str = self.gen_expr(func);
277                format!("{func_str}({})", arg_strs.join(", "))
278            }
279            IrExpr::Field { base, field } => {
280                let base_str = self.gen_expr(base);
281                format!("{base_str}.{field}")
282            }
283            IrExpr::Record { fields } => {
284                let field_strs: Vec<String> = fields
285                    .iter()
286                    .map(|(name, val)| {
287                        let val_str = self.gen_expr(val);
288                        format!("{name}: {val_str}")
289                    })
290                    .collect();
291                format!("{{ {} }}", field_strs.join(", "))
292            }
293            IrExpr::Path(segments) => segments.join("."),
294        }
295    }
296
297    fn gen_literal(&self, lit: &IrLiteral) -> String {
298        match lit {
299            IrLiteral::Bool(b) => b.to_string(),
300            IrLiteral::Str(s) => format!("\"{}\"", s.replace('\"', "\\\"")),
301            IrLiteral::U16(n) => n.to_string(),
302            IrLiteral::U32(n) => n.to_string(),
303            IrLiteral::U64(n) => n.to_string(),
304            IrLiteral::Int(n) => n.to_string(),
305            IrLiteral::Unit => "undefined".to_string(),
306        }
307    }
308
309    fn binop_to_ts(&self, op: &IrBinOp) -> &str {
310        match op {
311            IrBinOp::Add => "+",
312            IrBinOp::Sub => "-",
313            IrBinOp::Mul => "*",
314            IrBinOp::Div => "/",
315            IrBinOp::Mod => "%",
316            IrBinOp::Eq => "===",
317            IrBinOp::Ne => "!==",
318            IrBinOp::Lt => "<",
319            IrBinOp::Le => "<=",
320            IrBinOp::Gt => ">",
321            IrBinOp::Ge => ">=",
322            IrBinOp::And => "&&",
323            IrBinOp::Or => "||",
324        }
325    }
326
327    fn unaryop_to_ts(&self, op: &IrUnaryOp) -> &str {
328        match op {
329            IrUnaryOp::Neg => "-",
330            IrUnaryOp::Not => "!",
331            IrUnaryOp::Await => "await",
332        }
333    }
334
335    fn write_line(&mut self, line: &str) {
336        let indent = "  ".repeat(self.indent_level);
337        self.output.push_str(&indent);
338        self.output.push_str(line);
339        self.output.push('\n');
340    }
341}
342
343impl Default for TsCodegen {
344    fn default() -> Self {
345        Self::new()
346    }
347}
348
349/// Generate TypeScript code from IR module
350pub fn generate_typescript(module: &IrModule) -> String {
351    let mut codegen = TsCodegen::new();
352    codegen.generate(module)
353}
354
355/// Generate TypeScript code from IR module with optimization
356pub fn generate_typescript_optimized(
357    module: &IrModule,
358    opt_level: z1_ir::optimize::OptLevel,
359) -> String {
360    let mut optimized = module.clone();
361    z1_ir::optimize::optimize(&mut optimized, opt_level);
362    let mut codegen = TsCodegen::new();
363    codegen.generate(&optimized)
364}

370    #[test]
371    fn test_generate_simple_function() {
372        let module = IrModule {
373            name: "test".to_string(),
374            version: "1.0.0".to_string(),
375            imports: vec![],
376            types: vec![],
377            functions: vec![IrFunction {
378                name: "greet".to_string(),
379                params: vec![("name".to_string(), IrType::Str)],
380                return_type: IrType::Str,
381                effects: vec![],
382                body: IrBlock {
383                    statements: vec![IrStmt::Return {
384                        value: Some(IrExpr::Literal(IrLiteral::Str("Hello".to_string()))),
385                    }],
386                },
387            }],
388            exports: vec!["greet".to_string()],
389        };
390
391        let ts = generate_typescript(&module);
392        assert!(ts.contains("export function greet(name: string): string"));
393        assert!(ts.contains("return \"Hello\";"));
394        assert!(ts.contains("export { greet };"));
395    }
